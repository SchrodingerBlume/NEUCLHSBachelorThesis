% =========================== %
% ========  打印设置  ======== %
% =========================== %

% 打印模式可选参数：
% a. twoside：双面打印，会生成空白页
% b. oneside：单面打印，不会生成空白页
% 无论哪种，新的节都会被放置在奇数页
\def\PrintMode{twoside}

% 对称页边距可选开关：
% a. true：对称页边距
% b. false：固定页边距
\def\SymmetricMargins{false}

% 建议参数：
% a. 自己阅读查看的电子版：oneside+false
% b. 最终提交的电子版本：twoside+false
% c. 打印版本：twoside+true
% d. 折磨自己或他人：oneside+true
